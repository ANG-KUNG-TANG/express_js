/** * admin/users.js — extended with link-teacher functionality * * New: "Link to teacher" button on student rows opens a prompt * to enter the teacher's userId and calls * PATCH /api/admin/users/:studentId/link-teacher { teacherId } */ import { requireAdmin } from '../../core/router.js'; import { apiFetch } from '../../core/api.js'; import { initNavbar } from '../../../components/navbar.js'; import { toast } from '../../core/toast.js'; requireAdmin(); initNavbar(); const fmt = d => d ? new Date(d).toLocaleDateString('en-GB', { day:'numeric', month:'short', year:'numeric' }) : '—'; const roleBadge = r => { const cls = { admin:'badge--admin', teacher:'badge--teacher', user:'badge--default' }; return `${(r ?? '—').toUpperCase()}`; }; // ── Load all users ──────────────────────────────────────────────────────────── const loadUsers = async () => { const el = document.getElementById('users-list'); el.innerHTML = '

Loading…

'; try { const { data: users } = await apiFetch('/api/admin/users'); renderUsers(users ?? []); } catch (err) { el.innerHTML = '

Failed to load users.

'; toast(err.message, 'error'); } }; // ── Search by email ─────────────────────────────────────────────────────────── const searchByEmail = async () => { const email = document.getElementById('search-email').value.trim(); if (!email) { loadUsers(); return; } try { const { data: user } = await apiFetch(`/api/admin/users/email/${encodeURIComponent(email)}`); renderUsers(user ? [user] : []); } catch { document.getElementById('users-list').innerHTML = '

No user found with that email.

'; } }; // ── Render table ────────────────────────────────────────────────────────────── const renderUsers = (users) => { const el = document.getElementById('users-list'); if (!users.length) { el.innerHTML = '

No users found.

'; return; } el.innerHTML = ` ${users.map(u => { const uid = u.id ?? u._id; const role = u.role ?? u._role; const hasTeacher = !!(u.assignedTeacher); return ` `; }).join('')}
NameEmailRoleAssigned TeacherJoinedActions
${u.name ?? u._name} ${u.email} ${roleBadge(role)} ${hasTeacher ? `Linked` : `—`} ${fmt(u.createdAt ?? u._createdAt)} ${role !== 'admin' ? ` ` : ''} ${role === 'user' ? ` ` : ''} ${role === 'user' && hasTeacher ? ` ` : ''} ${role !== 'admin' ? ` ` : ''}
`; el.querySelectorAll('[data-action]').forEach(btn => btn.addEventListener('click', () => { const { action, id } = btn.dataset; if (action === 'promote') promoteUser(id); if (action === 'assign-teacher') assignTeacherRole(id); if (action === 'link-teacher') openLinkTeacherModal(id); if (action === 'unlink-teacher') unlinkTeacher(id); if (action === 'delete') deleteUser(id); }) ); }; // ── Actions ─────────────────────────────────────────────────────────────────── const promoteUser = async (id) => { if (!confirm('Promote to Admin? This cannot be undone.')) return; try { await apiFetch(`/api/admin/users/${id}/promote`, { method:'PATCH' }); toast('Promoted to Admin.'); loadUsers(); } catch (err) { toast(err.message, 'error'); } }; const assignTeacherRole = async (id) => { if (!confirm('Assign Teacher role to this user?')) return; try { await apiFetch(`/api/admin/users/${id}/assign-teacher`, { method:'PATCH' }); toast('Teacher role assigned.'); loadUsers(); } catch (err) { toast(err.message, 'error'); } }; const deleteUser = async (id) => { if (!confirm('Delete this user? Cannot be undone.')) return; try { await apiFetch(`/api/admin/users/${id}`, { method:'DELETE' }); toast('User deleted.'); loadUsers(); } catch (err) { toast(err.message, 'error'); } }; const unlinkTeacher = async (studentId) => { if (!confirm('Remove this student from their teacher?')) return; try { await apiFetch(`/api/admin/users/${studentId}/unlink-teacher`, { method:'PATCH' }); toast('Student unlinked from teacher.'); loadUsers(); } catch (err) { toast(err.message, 'error'); } }; // ── Link teacher modal ──────────────────────────────────────────────────────── const openLinkTeacherModal = (studentId) => { // Remove any existing modal document.getElementById('link-modal')?.remove(); const modal = document.createElement('div'); modal.id = 'link-modal'; modal.style.cssText = 'position:fixed;inset:0;background:rgba(0,0,0,.45);z-index:1000;display:flex;align-items:center;justify-content:center'; modal.innerHTML = `

Link student to a teacher

Enter the teacher's user ID. The student will then appear in that teacher's student list.

`; document.body.appendChild(modal); document.getElementById('link-cancel').addEventListener('click', () => modal.remove()); modal.addEventListener('click', e => { if (e.target === modal) modal.remove(); }); document.getElementById('link-confirm').addEventListener('click', async () => { const teacherId = document.getElementById('link-teacher-id').value.trim(); if (!teacherId) { toast('Please enter a teacher ID.', 'error'); return; } const btn = document.getElementById('link-confirm'); btn.disabled = true; btn.textContent = 'Linking…'; try { await apiFetch(`/api/admin/users/${studentId}/link-teacher`, { method: 'PATCH', body: JSON.stringify({ teacherId }), }); toast('Student linked to teacher successfully.'); modal.remove(); loadUsers(); } catch (err) { toast(err.message, 'error'); btn.disabled = false; btn.textContent = 'Link'; } }); }; // ── Event listeners ─────────────────────────────────────────────────────────── document.getElementById('search-btn').addEventListener('click', searchByEmail); document.getElementById('search-email').addEventListener('keydown', e => { if (e.key === 'Enter') searchByEmail(); }); document.getElementById('clear-search-btn').addEventListener('click', () => { document.getElementById('search-email').value = ''; loadUsers(); }); // ── Init ────────────────────────────────────────────────────────────────────── loadUsers();
